124//------------- Bytes -------------//
125TU_ATTR_ALWAYS_INLINE static inline uint32_t tu_u32(uint8_t b3, uint8_t b2, uint8_t b1, uint8_t b0)
126{
127 return ( ((uint32_t) b3) << 24) | ( ((uint32_t) b2) << 16) | ( ((uint32_t) b1) << 8) | b0;
128}
129
130TU_ATTR_ALWAYS_INLINE static inline uint16_t tu_u16(uint8_t high, uint8_t low)
131{
132 return (uint16_t) ((((uint16_t) high) << 8) | low);
133}

225#elif TUP_MCU_STRICT_ALIGN
226
227// MCU such as LPC_IP3511 Highspeed cannot access unaligned memory on USB_RAM although it is ARM M4.
228// We have to manually pick up bytes since tu_unaligned_uint32_t will still generate unaligned code
229// NOTE: volatile cast to memory to prevent compiler to optimize and generate unaligned code
230// TODO Big Endian may need minor changes
231TU_ATTR_ALWAYS_INLINE static inline uint32_t tu_unaligned_read32(const void* mem)
232{
233 volatile uint8_t const* buf8 = (uint8_t const*) mem;
234 return tu_u32(buf8[3], buf8[2], buf8[1], buf8[0]);
235}
236
237TU_ATTR_ALWAYS_INLINE static inline void tu_unaligned_write32(void* mem, uint32_t value)
238{
239 volatile uint8_t* buf8 = (uint8_t*) mem;
240 buf8[0] = tu_u32_byte0(value);
241 buf8[1] = tu_u32_byte1(value);
242 buf8[2] = tu_u32_byte2(value);
243 buf8[3] = tu_u32_byte3(value);
244}
245
246TU_ATTR_ALWAYS_INLINE static inline uint16_t tu_unaligned_read16(const void* mem)
247{
248 volatile uint8_t const* buf8 = (uint8_t const*) mem;
249 return tu_u16(buf8[1], buf8[0]);
250}
251
252TU_ATTR_ALWAYS_INLINE static inline void tu_unaligned_write16(void* mem, uint16_t value)
253{
254 volatile uint8_t* buf8 = (uint8_t*) mem;
255 buf8[0] = tu_u16_low(value);
256 buf8[1] = tu_u16_high(value);
257}
