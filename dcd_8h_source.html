50typedef enum
51{
52 DCD_EVENT_INVALID = 0,
53 DCD_EVENT_BUS_RESET,
54 DCD_EVENT_UNPLUGGED,
55 DCD_EVENT_SOF,
56 DCD_EVENT_SUSPEND, // TODO LPM Sleep L1 support

59 DCD_EVENT_SETUP_RECEIVED,

109// clean/flush data cache: write cache -> memory.
110// Required before an DMA TX transfer to make sure data is in memory
111void dcd_dcache_clean(void* addr, uint32_t data_size) TU_ATTR_WEAK;
112
113// invalidate data cache: mark cache as invalid, next read will read from memory
114// Required BOTH before and after an DMA RX transfer
115void dcd_dcache_invalidate(void* addr, uint32_t data_size) TU_ATTR_WEAK;

117// clean and invalidate data cache

199// helper to send bus signal event
200TU_ATTR_ALWAYS_INLINE static inline void dcd_event_bus_signal (uint8_t rhport, dcd_eventid_t eid, bool in_isr)
201{
202 dcd_event_t event = { .rhport = rhport, .event_id = eid };
203 dcd_event_handler(&event, in_isr);
204}

206// helper to send bus reset event
207TU_ATTR_ALWAYS_INLINE static inline void dcd_event_bus_reset (uint8_t rhport, tusb_speed_t speed, bool in_isr)
208{
209 dcd_event_t event = { .rhport = rhport, .event_id = DCD_EVENT_BUS_RESET };
210 event.bus_reset.speed = speed;
211 dcd_event_handler(&event, in_isr);
212}
213
214// helper to send setup received
215TU_ATTR_ALWAYS_INLINE static inline void dcd_event_setup_received(uint8_t rhport, uint8_t const * setup, bool in_isr)
216{
217 dcd_event_t event = { .rhport = rhport, .event_id = DCD_EVENT_SETUP_RECEIVED };
218 memcpy(&event.setup_received, setup, sizeof(tusb_control_request_t));
219
220 dcd_event_handler(&event, in_isr);
221}
222
223// helper to send transfer complete event
224TU_ATTR_ALWAYS_INLINE static inline void dcd_event_xfer_complete (uint8_t rhport, uint8_t ep_addr, uint32_t xferred_bytes, uint8_t result, bool in_isr)
225{
226 dcd_event_t event = { .rhport = rhport, .event_id = DCD_EVENT_XFER_COMPLETE };
227
228 event.xfer_complete.ep_addr = ep_addr;
229 event.xfer_complete.len = xferred_bytes;
230 event.xfer_complete.result = result;
231
232 dcd_event_handler(&event, in_isr);
233}
