507// Get direction from Endpoint address
508TU_ATTR_ALWAYS_INLINE static inline tusb_dir_t tu_edpt_dir(uint8_t addr)
509{
510 return (addr & TUSB_DIR_IN_MASK) ? TUSB_DIR_IN : TUSB_DIR_OUT;
511}
512
513// Get Endpoint number from address
514TU_ATTR_ALWAYS_INLINE static inline uint8_t tu_edpt_number(uint8_t addr)
515{
516 return (uint8_t)(addr & (~TUSB_DIR_IN_MASK));
517}
518
519TU_ATTR_ALWAYS_INLINE static inline uint8_t tu_edpt_addr(uint8_t num, uint8_t dir)
520{
521 return (uint8_t)(num | (dir ? TUSB_DIR_IN_MASK : 0));
522}
523
524TU_ATTR_ALWAYS_INLINE static inline uint16_t tu_edpt_packet_size(tusb_desc_endpoint_t const* desc_ep)
525{
526 return tu_le16toh(desc_ep->wMaxPacketSize) & TU_GENMASK(10, 0);
527}

547// return next descriptor
548TU_ATTR_ALWAYS_INLINE static inline uint8_t const * tu_desc_next(void const* desc)
549{
550 uint8_t const* desc8 = (uint8_t const*) desc;
551 return desc8 + desc8[DESC_OFFSET_LEN];
552}
553
554// get descriptor type
555TU_ATTR_ALWAYS_INLINE static inline uint8_t tu_desc_type(void const* desc)
556{
557 return ((uint8_t const*) desc)[DESC_OFFSET_TYPE];
558}
559
560// get descriptor length
561TU_ATTR_ALWAYS_INLINE static inline uint8_t tu_desc_len(void const* desc)
562{
563 return ((uint8_t const*) desc)[DESC_OFFSET_LEN];
564}
